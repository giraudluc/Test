The TelemetryPlot.py Python module provides a set of services to plot telemetry data. Telemetry data are sent over UDP Packets in a format compatible with the Teleplot for VSCode extension.

def initializeTelemetry() -> bool:
  """Initialize the telemetry.
  Create an UDP socket to communicate with Teleplot.

  Returns:
    bool: Initialization ok or not.
  """
  ok = False
  sock = _createUdpSocket()
  if(sock != None and type(sock) == socket.socket):
    ok = True
  return(ok)

def sendTelemetryText(chanelName: str, text: str):
  """Send the given text message to the given Telemetry chanel name.

  Args:
    chanelName (str): Telemetry chanel name
    text (str): Text message
  """
  msg = "%s:%s|t" % (chanelName, text)
  _sendTelemetryMessage(msg)

def sendTelemetryTimeStampedValue(chanelName: str, value: Union[int, float]):
  """Send the given value to the given Telemetry chanel name.

  Args:
    chanelName (str): Telemetry chanel name
    value (Union[int, float]): Value to send
  """
  timeStamp = _getTimeStamp()
  msg = "%s:%s:%s|g" % (chanelName, str(timeStamp), str(value))
  _sendTelemetryMessage(msg)

def sendTelemetryXY(chanelName: str, xArray: List[Union[int, float]], yArray: List[Union[int, float]]):
  """Send the given x and y values to the given Telemetry chanel name.

  Args:
    chanelName (str): Telemetry chanel name
    xArray (List[Union[int, float]]): x values array
    yArray (List[Union[int, float]]): y values array
  """
  # Loop
  for i in range(len(xArray)):
    x = xArray[i]
    y = yArray[i]
    # message
    msg = "%s:%s:%s|xy" % (chanelName, str(x), str(y))
    _sendTelemetryMessage(msg)
